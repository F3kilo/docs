This guide covers best practices, patterns, and recommendations for building applications with the Drakula API.

<Step title="Create a Configuration Module">

```python
# config.py
import os
from dotenv import load_dotenv

load_dotenv()

class Config:
    API_TOKEN = os.getenv("DRAKULA_API_TOKEN")
    BASE_URL = os.getenv("DRAKULA_API_URL", "https://api.drakula.com/api/v1")
    TIMEOUT = 30  # seconds
    MAX_RETRIES = 3
    
    @classmethod
    def validate(cls):
        if not cls.API_TOKEN:
            raise ValueError("DRAKULA_API_TOKEN environment variable is required")
```

</Step>

<Step title="Implement API Client">

```python
# api_client.py
import requests
import time
from typing import Optional, Dict, Any
from config import Config

class DrakulaAPIClient:
    def __init__(self):
        Config.validate()
        self.base_url = Config.BASE_URL
        self.headers = {
            "Authorization": f"Bearer {Config.API_TOKEN}",
            "Content-Type": "application/json",
            "User-Agent": "Drakula-API-Client/1.0.0"
        }
        self.session = requests.Session()
        self.session.headers.update(self.headers)
    
    def _make_request(self, method: str, endpoint: str, 
                     data: Optional[Dict] = None, 
                     params: Optional[Dict] = None,
                     retry_count: int = 0) -> Dict[str, Any]:
        url = f"{self.base_url}/{endpoint}"
        
        try:
            response = self.session.request(
                method=method,
                url=url,
                json=data,
                params=params,
                timeout=Config.TIMEOUT
            )
            response.raise_for_status()
            return response.json()
            
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 429 and retry_count < Config.MAX_RETRIES:
                # Exponential backoff for rate limiting
                wait_time = (2 ** retry_count) + 1
                print(f"Rate limited. Retrying in {wait_time} seconds...")
                time.sleep(wait_time)
                return self._make_request(method, endpoint, data, params, retry_count + 1)
            else:
                raise
    
    def get_user_profile(self):
        return self._make_request("GET", "user/profile")
    
    def get_token_details(self, chain: str, token_address: str):
        return self._make_request("GET", f"token/{chain}/{token_address}/details")
    
    # Add more methods as needed
```

</Step>
</Steps>

## 3. Error Handling Patterns

### Comprehensive Error Handler

```python
# error_handler.py
from typing import Dict, Any, Optional
import requests

class APIError(Exception):
    def __init__(self, message: str, status_code: Optional[int] = None, 
                 error_data: Optional[Dict] = None):
        self.message = message
        self.status_code = status_code
        self.error_data = error_data
        super().__init__(self.message)

def handle_api_error(error: requests.exceptions.RequestException) -> None:
    """Handle API errors with specific messaging."""
    
    if isinstance(error, requests.exceptions.HTTPError):
        status_code = error.response.status_code
        
        if status_code == 401:
            raise APIError(
                "Authentication failed. Please check your API token.",
                status_code=401
            )
        elif status_code == 403:
            raise APIError(
                "Insufficient permissions for this action.",
                status_code=403
            )
        elif status_code == 429:
            raise APIError(
                "Rate limit exceeded. Please wait before making more requests.",
                status_code=429
            )
        elif 400 <= status_code < 500:
            # Try to parse error message from response
            try:
                error_data = error.response.json()
                error_msg = error_data.get("error", {}).get("message", "Client error")
            except:
                error_msg = "Client error"
            raise APIError(error_msg, status_code=status_code, error_data=error_data)
        elif status_code >= 500:
            raise APIError(
                "Server error. Please try again later or contact support.",
                status_code=status_code
            )
    
    elif isinstance(error, requests.exceptions.Timeout):
        raise APIError("Request timed out. Please try again.")
    
    elif isinstance(error, requests.exceptions.ConnectionError):
        raise APIError("Network connection error. Please check your internet connection.")
    
    else:
        raise APIError(f"Request failed: {str(error)}")
```

## 4. Performance Optimization

### Caching Strategy

```python
# cache_manager.py
import time
from typing import Dict, Any, Optional
from functools import lru_cache

class APICache:
    def __init__(self, ttl: int = 300):  # 5 minutes default TTL
        self.ttl = ttl
        self.cache: Dict[str, Dict[str, Any]] = {}
    
    def get(self, key: str) -> Optional[Dict[str, Any]]:
        if key in self.cache:
            entry = self.cache[key]
            if time.time() - entry["timestamp"] < self.ttl:
                return entry["data"]
            else:
                del self.cache[key]
        return None
    
    def set(self, key: str, data: Dict[str, Any]) -> None:
        self.cache[key] = {
            "data": data,
            "timestamp": time.time()
        }

# Usage with API client
cache = APICache(ttl=60)  # 1 minute cache for market data

@lru_cache(maxsize=128)
def get_cached_token_details(chain: str, token_address: str):
    cached = cache.get(f"token:{chain}:{token_address}")
    if cached:
        return cached
    
    # Make API call
    client = DrakulaAPIClient()
    result = client.get_token_details(chain, token_address)
    cache.set(f"token:{chain}:{token_address}", result)
    return result
```

### Batch Processing

```python
# batch_processor.py
from typing import List, Dict, Any
import asyncio
import aiohttp
from config import Config

class BatchProcessor:
    def __init__(self, max_concurrent: int = 10):
        self.max_concurrent = max_concurrent
    
    async def fetch_multiple_tokens(self, token_list: List[Dict]) -> List[Dict[str, Any]]:
        """Fetch multiple tokens concurrently."""
        semaphore = asyncio.Semaphore(self.max_concurrent)
        
        async def fetch_token(session: aiohttp.ClientSession, token: Dict):
            async with semaphore:
                url = f"{Config.BASE_URL}/token/{token['chain']}/{token['address']}/details"
                headers = {"Authorization": f"Bearer {Config.API_TOKEN}"}
                
                async with session.get(url, headers=headers) as response:
                    if response.status == 200:
                        data = await response.json()
                        return {**token, "data": data}
                    else:
                        return {**token, "error": f"HTTP {response.status}"}
        
        async with aiohttp.ClientSession() as session:
            tasks = [fetch_token(session, token) for token in token_list]
            return await asyncio.gather(*tasks, return_exceptions=True)
```

## 5. Security Best Practices

### Secure Token Management

```python
# security.py
import os
import keyring
from cryptography.fernet import Fernet
import base64

class SecureTokenManager:
    def __init__(self, service_name: str = "drakula_api"):
        self.service_name = service_name
        self.key = self._get_or_create_key()
        self.cipher = Fernet(self.key)
    
    def _get_or_create_key(self) -> bytes:
        """Get encryption key from keyring or create new one."""
        key = keyring.get_password("system", f"{self.service_name}_key")
        if key:
            return base64.urlsafe_b64decode(key)
        else:
            new_key = Fernet.generate_key()
            keyring.set_password("system", f"{self.service_name}_key", 
                               base64.urlsafe_b64encode(new_key).decode())
            return new_key
    
    def store_token(self, token: str, username: str = "default"):
        """Encrypt and store API token."""
        encrypted = self.cipher.encrypt(token.encode())
        keyring.set_password(self.service_name, username, encrypted.decode())
    
    def get_token(self, username: str = "default") -> str:
        """Retrieve and decrypt API token."""
        encrypted = keyring.get_password(self.service_name, username)
        if not encrypted:
            raise ValueError(f"No token found for user: {username}")
        decrypted = self.cipher.decrypt(encrypted.encode())
        return decrypted.decode()
```

## 6. Monitoring and Logging

### Structured Logging

```python
# logging_config.py
import logging
import json
from datetime import datetime
from typing import Dict, Any

class JSONFormatter(logging.Formatter):
    def format(self, record: logging.LogRecord) -> str:
        log_data = {
            "timestamp": datetime.utcnow().isoformat() + "Z",
            "level": record.levelname,
            "message": record.getMessage(),
            "module": record.module,
            "function": record.funcName,
            "line": record.lineno,
        }
        
        if hasattr(record, "api_data"):
            log_data["api_data"] = record.api_data
        
        if record.exc_info:
            log_data["exception"] = self.formatException(record.exc_info)
        
        return json.dumps(log_data)

def setup_logging():
    logger = logging.getLogger("drakula_api")
    logger.setLevel(logging.INFO)
    
    # Console handler with JSON format
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(JSONFormatter())
    logger.addHandler(console_handler)
    
    # File handler
    file_handler = logging.FileHandler("api_logs.json")
    file_handler.setFormatter(JSONFormatter())
    logger.addHandler(file_handler)
    
    return logger

# Usage
logger = setup_logging()

def log_api_call(endpoint: str, method: str, status_code: int, duration: float):
    logger.info(
        "API call completed",
        extra={
            "api_data": {
                "endpoint": endpoint,
                "method": method,
                "status_code": status_code,
                "duration_ms": duration * 1000
            }
        }
    )
```
